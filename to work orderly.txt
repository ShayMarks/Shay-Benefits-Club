project-root/
├── api/
│   ├── models/
│   │   ├── user.js        # מודל לניהול משתמשים
│   │   ├── product.js     # מודל לניהול מוצרים
│   │   └── category.js    # מודל לניהול קטגוריות
│   ├── routes/
│   │   ├── userRoutes.js  # ניתובים לפעולות על משתמשים
│   │   ├── productRoutes.js # ניתובים לפעולות על מוצרים
│   │   └── categoryRoutes.js # ניתובים לפעולות על קטגוריות
│   └── controllers/
│       ├── userController.js # בקרים לניהול משתמשים
│       ├── productController.js # בקרים לניהול מוצרים
│       └── categoryController.js # בקרים לניהול קטגוריות
├── views/
│   ├── layouts/
│   │   ├── main.hbs       # תבנית עיקרית לכל הדפים
│   ├── partials/
│   │   ├── footer.hbs     # חלקי תחתית להכללה בדפים
│   │   ├── footer2.hbs    # עוד חלקי תחתית לדוגמה
│   │   └── toolbar/
│   │       ├── adminToolBar.hbs # הסרגל כלים שיופיע אחרי התבנית בדף האדמין
│   │       ├── userToolBar.hbs # הסרגל כלים שיופיע אחרי התבנית בשאר הדפים
│   │       └── shopToolBar.hbs # הסרגיל כלים שיופיע בדף החנות אחרי הסרגל כלים הרגיל
│   ├── home.hbs           # דף הבית
│   ├── contact.hbs        # דף יצירת קשר
│   ├── login.hbs          # דף התחברות
│   ├── register.hbs       # דף הרשמה
│   └── shop.hbs           # דף החנות
├── public/
│   ├── css/
│   │   └── styles.css     # קבצי סגנון
│   └── js/
│       └── script.js      # סקריפטים לצד הלקוח
├── middleware/
│   └── auth.js            # תוכנית ביניים לאימות
├── app.js                 # הגדרות והתחלה של אפליקציה
├── server.js              # הגדרות שרת והאזנה לפורט
└── .env                   # קובץ להגדרות סביבה (מחרוזות חיבור וכו')
